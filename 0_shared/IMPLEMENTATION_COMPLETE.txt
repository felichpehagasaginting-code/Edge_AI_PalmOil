## ✨ USAGE EXAMPLE

```python
# Setup (startup)
from logger_config import setup_logger, get_logger
from error_tracker import ErrorTracker, ErrorCategory, ErrorSeverity
from performance_monitor import PerformanceMonitor

setup_logger(app_name="my_service")
logger = get_logger(__name__)
tracker = ErrorTracker(logger=logger)
monitor = PerformanceMonitor(logger=logger)

# Use in code
with monitor.track_operation("api_call", component="api"):
    try:
        result = await api.fetch_data()
        logger.info_structured("Data fetched", items=len(result))
    except TimeoutError as e:
        tracker.track_error(
            category=ErrorCategory.API,
            severity=ErrorSeverity.ERROR,
            message=str(e),
            error_code="API_TIMEOUT",
            component="api",
        )
        raise

# Query status
stats = tracker.get_error_stats(minutes=60)
health = tracker.get_component_health("api")
sla = monitor.check_sla("api_call", max_latency_ms=200)
```
